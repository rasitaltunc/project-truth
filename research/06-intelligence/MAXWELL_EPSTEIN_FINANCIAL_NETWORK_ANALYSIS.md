    for window in windows:

        num_transactions = len(window.transactions)
        individual_amounts = [t.amount for t in window.transactions]

        # Check for structuring indicators

def infer_beneficial_owner(entity_dict):
    """
    Multi-factor beneficial ownership determination
    Returns: (beneficial_owner, confidence_0_to_100)
    """

    scores = {}

    # Factor 1: Account signatory
    if entity_dict.get('primary_signer'):
        primary = entity_dict['primary_signer']
        scores[primary] = scores.get(primary, 0) + 30

    # Factor 2: Fund source
    if entity_dict.get('fund_source_person'):
        source = entity_dict['fund_source_person']
        scores[source] = scores.get(source, 0) + 25

    # Factor 3: Fund destination
    if entity_dict.get('fund_destination_person'):
        dest = entity_dict['fund_destination_person']
        scores[dest] = scores.get(dest, 0) + 20

    # Factor 4: Control authority
    if entity_dict.get('decision_maker'):
        maker = entity_dict['decision_maker']
        scores[maker] = scores.get(maker, 0) + 15

    # Factor 5: Legal title holder
    if entity_dict.get('legal_owner'):
        owner = entity_dict['legal_owner']
        scores[owner] = scores.get(owner, 0) + 10

    # Find highest-scoring person
    if not scores:
        return ('UNKNOWN', 0)

    beneficial_owner = max(scores, key=scores.get)
    confidence = int((scores[beneficial_owner] / 100) * 100)  # Normalize to 0-100

    return (beneficial_owner, confidence)

def classify_laundering_stage(transaction_context):
    """
    Classify transaction as likely PLACEMENT, LAYERING, or INTEGRATION
    """

    # PLACEMENT indicators
    if transaction_context['entry_to_financial_system'] and \

        stage_scores['placement'] += 30

    if transaction_context['cash_deposit']:

        stage_scores['placement'] += 15

       transaction_context['complexity'] > 2:
        stage_scores['layering'] += 30

    if transaction_context['overseas_routing']:
        stage_scores['layering'] += 25

    if transaction_context['shell_company_involved']:

    if transaction_context['no_legitimate_business_purpose']:
        stage_scores['layering'] += 15
